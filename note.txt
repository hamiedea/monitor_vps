# 1. Persiapan Server (dijalankan sekali saja)
sudo apt-get update
sudo apt-get upgrade -y
sudo apt-get install python3-pip screen -y

# 2. Buat direktori dan masuk
mkdir monitor_vps
cd monitor_vps

# 3. Buat file config.json (tempel isi config Anda di sini)
nano config.json

# 4. Buat file bot.py (tempel kode bot Anda di sini)
nano bot.py

# 5. Instal library yang dibutuhkan (dengan versi yang benar)
pip3 install "python-telegram-bot[job-queue]"

# 6. Jalankan bot di dalam 'screen' agar tidak mati saat terminal ditutup
screen -S bot_monitor

# 7. jalankan bot
python3 bot.py
